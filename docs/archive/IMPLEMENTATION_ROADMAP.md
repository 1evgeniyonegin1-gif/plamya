async def send_scheduled_reminders(bot: Bot):
    """–û—Ç–ø—Ä–∞–≤–ª—è–µ—Ç –∑–∞–ø–ª–∞–Ω–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ –Ω–∞–ø–æ–º–∏–Ω–∞–Ω–∏—è"""

        # –ù–∞—Ö–æ–¥–∏–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π, –∫–æ—Ç–æ—Ä—ã–º –Ω—É–∂–Ω–æ –æ—Ç–ø—Ä–∞–≤–∏—Ç—å –Ω–∞–ø–æ–º–∏–Ω–∞–Ω–∏–µ

        result = await session.execute(
            select(User).where(
                and_(
                    User.funnel_step > 0,
                    User.lead_status.in_(["new", "warming"]),
                    User.funnel_started_at.isnot(None)
                )
            )
        )
        users = result.scalars().all()

        for user in users:
            # –ü–æ–ª—É—á–∞–µ–º —Ü–µ–ø–æ—á–∫—É –¥–ª—è —ç—Ç–æ–≥–æ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
            sequence = get_sequence(
                user.user_intent,
                user.pain_point,
                user.income_goal
            )
            if not sequence:
                continue

            # –ù–∞—Ö–æ–¥–∏–º —Å–ª–µ–¥—É—é—â–µ–µ –∞–≤—Ç–æ—Å–æ–æ–±—â–µ–Ω–∏–µ
            for i, step in enumerate(sequence):
                if not step.get("auto"):
                    continue

                delay_hours = step.get("delay_hours", 24)
                send_time = user.funnel_started_at + timedelta(hours=delay_hours)

                # –ü—Ä–æ–≤–µ—Ä—è–µ–º, –ø–æ—Ä–∞ –ª–∏ –æ—Ç–ø—Ä–∞–≤–ª—è—Ç—å
                if send_time <= now:

async def send_reminder(bot: Bot, user: User, step: dict):
    """–û—Ç–ø—Ä–∞–≤–ª—è–µ—Ç –æ–¥–Ω–æ –Ω–∞–ø–æ–º–∏–Ω–∞–Ω–∏–µ"""
    message = step["message"].format(
        first_name=user.first_name or "–î—Ä—É–≥"
    )

    # –§–æ—Ä–º–∏—Ä—É–µ–º –∫–ª–∞–≤–∏–∞—Ç—É—Ä—É
    keyboard = None
    if step.get("buttons"):
        from aiogram.types import InlineKeyboardMarkup, InlineKeyboardButton
        buttons = []
        for btn in step["buttons"]:
            if btn.get("url"):
                buttons.append([InlineKeyboardButton(
                    text=btn["text"],
                    url=btn["url"]
                )])
            else:
                buttons.append([InlineKeyboardButton(
                    text=btn["text"],
                    callback_data=btn["callback"]
                )])
        keyboard = InlineKeyboardMarkup(inline_keyboard=buttons)

    try:
        await bot.send_message(
            chat_id=user.telegram_id,
            text=message,
            reply_markup=keyboard
        )
    except Exception as e:
        print(f"–û—à–∏–±–∫–∞ –æ—Ç–ø—Ä–∞–≤–∫–∏ –Ω–∞–ø–æ–º–∏–Ω–∞–Ω–∏—è {user.telegram_id}: {e}")

        'interval',
        hours=1,  # –ü—Ä–æ–≤–µ—Ä—è–µ–º –∫–∞–∂–¥—ã–π —á–∞—Å
        args=[bot],
        id='send_reminders'

    scheduler.start()

            user.nl_partner_id = nl_partner_id
            await session.commit()
